拼接多个DataFrame

# 按列拼接
df = pd.concat([df1, df2], axis=1)

# 按行拼接
df = pd.concat([df1, df2], axis=0)

# 按行拼接，忽略索引
df = pd.concat([df1, df2], axis=0, ignore_index=True)

DataFrame转换数据类型

df = df.astype("float32")
